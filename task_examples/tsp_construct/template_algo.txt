import numpy as np


def select_next_node(
    current_node: int,
    destination_node: int,
    unvisited_nodes: np.ndarray,
    distance_matrix: np.ndarray,
) -> int:
    """Select the next node in a constructive TSP tour.

    Args:
        current_node: Index of the current node.
        destination_node: Index of the fixed destination node.
        unvisited_nodes: Array of node indices that have not been visited yet.
        distance_matrix: Pairwise Euclidean distance matrix.

    Returns:
        The index of the next node to visit.
    """
    next_node = unvisited_nodes[0]
    return next_node
